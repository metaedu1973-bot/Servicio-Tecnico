let corredores = []; function agregarCorredor() { let numero = document.getElementById("numero").value; let nombre = document.getElementById("nombre").value; if (numero === "" || nombre === "") { alert("Ingresa número y nombre del corredor"); return; } corredores.push({ numero: numero, nombre: nombre, vueltas: 0 }); document.getElementById("numero").value = ""; document.getElementById("nombre").value = ""; mostrarCorredores(); } function sumarVuelta(index) { corredores[index].vueltas++; mostrarCorredores(); } function mostrarCorredores() { let lista = document.getElementById("lista"); lista.innerHTML = ""; corredores.forEach((corredor, index) => { lista.innerHTML += `

#${corredor.numero}

${corredor.nombre}

Vueltas: ${corredor.vueltas}

`; }); }
