let corredores = []; function agregarCorredor(){ let numero = document.getElementById("numero").value; let nombre = document.getElementById("nombre").value; corredores.push({ numero: numero, nombre: nombre, vueltas: 0 }); mostrarCorredores(); } function sumarVuelta(index){ corredores[index].vueltas++; mostrarCorredores(); } function mostrarCorredores(){ let lista = document.getElementById("lista"); lista.innerHTML = ""; corredores.forEach((corredor, index) => { lista.innerHTML += `

#${corredor.numero}

${corredor.nombre}

${corredor.vueltas}

`; }); }
